1. TO CALL LIBRARIES FOR USING THEIR FUNCTIONS

> library(AppliedPredictiveModeling)

> library(caret)

 

2. TO PREPARE DATA

> training<-read.csv("pml-training.csv")

> testing<-read.csv("pml-testing.csv")

 

1.      TO SPECIFY PREDICTORS BECAUSE OF TOO MANY VARIABLES

AND SOME ARE MISSING VARIBALES

> trainclear <-training[,c(7:10,160)]

 

4. TO APPLY CART MODEL FOR OUTCOMES (RANDOM FOREST) AS QUALITATIVE MULTI-CHOICE VARIABLES

> set.seed(32343)

> modelFit <-train(trainclear$classe~., data = trainclear,

 method="rf")
